def add_shot_event(self, x, y, shot_type, popup):
    entry = (len(self.log_entries) + 1, "S", shot_type, x, y, self.period_selected.get())  # Lägg till alla värden
    self.log_entries.append(entry)
    self.update_shot_log_treeview()  # Uppdatera Shot Log
    self.update_stats()  # Uppdatera statistik
    self.update_plot()  # Uppdatera plotten
    popup.destroy()

def add_goal_event(self, x, y, goal_type, popup):
    # Lägg till en post med 6 värden: (index, S/G, typ, X, Y, period)
    entry = (
        len(self.log_entries) + 1,  # # (index) - Räknar ut index baserat på längden på loggen
        "G",  # S/G (G för mål)
        goal_type,  # Type (t.ex., "goal", "breakaway" etc.)
        x,  # X position
        y,  # Y position
        self.period_selected.get()  # Period (t.ex., "Period 1", "All" etc.)
    )
    self.log_entries.append(entry)
    
    self.update_shot_log_treeview()  # Uppdatera Shot Log
    self.update_stats()  # Uppdatera statistik
    self.update_plot()  # Uppdatera plotten
    popup.destroy()
